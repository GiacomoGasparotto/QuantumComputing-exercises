Perform runtime debugging and validation checkpoints. 

The \textquotesingle{}checkpoint\textquotesingle{} subroutine provides an interface for handling debugging messages and validity checks for matrices during execution. Depending on the selected verbosity level, it can\+:
\begin{DoxyItemize}
\item 0\+: print a debug message
\item 1\+: check matrix dimension compatibility
\item 2\+: verify that a matrix is square
\end{DoxyItemize}


\begin{DoxyParams}[1]{Parameters}
\mbox{\texttt{in}}  & {\em debug} & Logical flag to enable or disable debugging. \\
\hline
\mbox{\texttt{in}}  & {\em verbosity} & Integer verbosity level\+:
\begin{DoxyItemize}
\item {\bfseries{0}}\+: print a debug message
\item {\bfseries{1}}\+: check matrix dimension compatibility
\item {\bfseries{2}}\+: verify if a matrix is square or not 
\end{DoxyItemize}\\
\hline
 & {\em \mbox{[}in,optional\mbox{]}} & message Custom message to print (verbosity 0 or 2). \\
\hline
 & {\em \mbox{[}in,optional\mbox{]}} & A(\+:,\+:) First matrix to check (verbosity 1). \\
\hline
 & {\em \mbox{[}in,optional\mbox{]}} & B(\+:,\+:) Second matrix to check (verbosity 1). \\
\hline
 & {\em \mbox{[}in,optional\mbox{]}} & nrows Number of rows (verbosity 2). \\
\hline
 & {\em \mbox{[}in,optional\mbox{]}} & ncols Number of columns (verbosity 2).\\
\hline
\end{DoxyParams}

\begin{DoxyExceptions}{Exceptions}
{\em Stops} & execution if\+:
\begin{DoxyItemize}
\item matrix dimensions are incompatible (verbosity=1)
\item matrix is not square (verbosity=2)
\item an invalid verbosity level is provided
\end{DoxyItemize}\\
\hline
\end{DoxyExceptions}
